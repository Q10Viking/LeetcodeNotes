2-Add-Two-Numbers

原题

来一个数据，我们就分配一个空间存储起来。

java

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode p1 = l1,p2 = l2;
        ListNode dumpyHead = new ListNode(0);
        ListNode cur  = dumpyHead;
        int carry=0;
        while(p1 != null && p2 != null){

            int sum = p1.val+p2.val+carry;
            cur.next = new ListNode(sum%10);
            //update 
            carry = sum/10;

            cur = cur.next;
            p1 = p1.next;
            p2 = p2.next;

        }
        //check 
        while(p1 != null){
            int sum =  p1.val+carry;
            carry = sum/10;
            cur.next = new ListNode(sum % 10);
            cur = cur.next;
            p1 = p1.next;
        }

        while(p2 != null){
            int sum =  p2.val+carry;
            carry = sum/10;
            cur.next = new ListNode(sum % 10);
            cur = cur.next;
            p2 = p2.next;
        }

        if(carry == 1){
            cur.next = new ListNode(carry);
        }

         //直接返回我们需要链表的形式，去掉开头的0
        return dumpyHead.next;

    }

}

在处理p1,p2是否还有剩余的时候，这段代码看起来非常的冗余

Java 优化

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode p1 = l1,p2 = l2;
        ListNode dumpyHead = new ListNode(0);
        ListNode cur  = dumpyHead;
        int carry=0;
        while(p1 != null || p2 != null){

            int x = (p1==null)?0:p1.val;
            int y = (p2==null)?0:p2.val;
            int sum = x+y+carry;

            carry = sum/10;
            cur.next = new ListNode(sum%10);

            //update for the next time
            if(p1 != null)
                p1 = p1.next;
            if(p2 != null)
                p2 = p2.next;
            cur = cur.next;      
        }

        if(carry == 1){
            cur.next = new ListNode(carry);
        }

         //直接返回我们需要链表的形式，去掉开头的0
        return dumpyHead.next;

    }

}
